    function dijkstra(graph, start):
    dist[*] = ∞
    dist[start] = 0
    pq = MinHeap()
    pq.push(0, start)
    while pq not empty:
        (d, u) = pq.pop()
        if d > dist[u]: continue
        for (v, w) in graph.neighbors(u):
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                pq.push(dist[v], v)
    return dist
